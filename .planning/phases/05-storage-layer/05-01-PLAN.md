    Section `# --- LocalStorageWriter ---`

    **LocalStorageWriter(BaseStorageWriter)**:

    Constructor:
    ```python
    def __init__(self, root_dir: str | None = None):
        """Initialize with optional root directory.

        Args:
            root_dir: Optional base directory (defaults to None — all paths absolute)
        """
        self.root_dir = Path(root_dir) if root_dir else None
    ```

    Helper method (not async):
    ```python
    def _extract_path(self, uri: str) -> Path:
        """Extract file path from file:// URI.

        Args:
            uri: Full URI (e.g., 'file:///tmp/x' or 'file://localhost/tmp/x')

        Returns:
            pathlib.Path

        Raises:
            URINotSupported: If URI doesn't start with 'file://'
        """
        if not uri.startswith("file://"):
            raise URINotSupported(f"Unsupported URI prefix: {uri}")
        # Remove file:// prefix, handle both file:///path and file://localhost/path
        path_str = uri[7:]  # Remove 'file://'
        if path_str.startswith("localhost/"):
            path_str = "/" + path_str[10:]  # Remove 'localhost' prefix
        return Path(path_str)
    ```

    Methods:

    ```python
    async def read_file(self, uri: str) -> bytes:
        """Read file via file:// URI."""
        path = self._extract_path(uri)
        if not path.is_file():  # False for dirs and nonexistent
            raise FileNotFound(f"File not found: {uri}")
        # Use asyncio.to_thread for I/O
        return await asyncio.to_thread(path.read_bytes)
    ```

    ```python
    async def write_file(self, uri: str, data: bytes) -> None:
        """Write file via file:// URI, creating parent directories."""
        path = self._extract_path(uri)
        parent = path.parent
        # Create parent dirs
        await asyncio.to_thread(lambda: parent.mkdir(parents=True, exist_ok=True))
        # Write file
        await asyncio.to_thread(path.write_bytes, data)
    ```

    ```python
    async def file_exists(self, uri: str) -> bool:
        """Check if file exists (False for directories)."""
        path = self._extract_path(uri)
        return await asyncio.to_thread(path.is_file)  # Returns False for dirs
    ```

    ```python
    async def delete_file(self, uri: str) -> None:
        """Delete file (idempotent)."""
        path = self._extract_path(uri)
        def _delete():
            if path.exists():
                path.unlink()
        await asyncio.to_thread(_delete)
    ```

    Constructor:

    def __init__(self):

    Config loading:
    ```python
    def _load_config(self) -> dict[str, str]:
        """Load STORAGE_BACKENDS from env or use defaults.

        Returns:
            Dict mapping URI prefix (e.g., 'file://') to writer class name
        """
        env_val = os.getenv("STORAGE_BACKENDS", "")
        if not env_val:
            return self.DEFAULT_BACKENDS

        try:
            config = json.loads(env_val)
            return config
        except json.JSONDecodeError:
            # Log warning, fall back to defaults (or raise — your choice)
            # For now, fall back silently
            return self.DEFAULT_BACKENDS
    ```

    Dispatch method:
    ```python
    def get_writer(self, uri: str) -> BaseStorageWriter:
        """Get writer instance for URI based on prefix.

        Args:
            uri: Full URI (e.g., 'file:///tmp/x')

        Returns:
            BaseStorageWriter instance

        Raises:
            URINotSupported: If URI prefix not in config
        """
        # Extract prefix (e.g., 'file://' from 'file:///tmp/x')
        prefix = self._extract_prefix(uri)

        if prefix not in self._backend_config:
            raise URINotSupported(f"Unsupported URI prefix: {prefix}")

        # Return cached or create new
        if prefix not in self._writers:
            writer_class_name = self._backend_config[prefix]
            writer_class = self._get_writer_class(writer_class_name)
            self._writers[prefix] = writer_class()

        return self._writers[prefix]
    ```

    Helpers:
    ```python
    def _extract_prefix(self, uri: str) -> str:
        """Extract URI prefix (e.g., 'file://' from 'file:///path')."""
        if "://" not in uri:
            raise URINotSupported(f"Invalid URI format: {uri}")
        return uri.split("://")[0] + "://"

    def _get_writer_class(self, class_name: str):
        """Dynamically load writer class from master.storage modules.

        Args:
            class_name: Class name (e.g., 'LocalStorageWriter')

        Returns:
            Class reference

        Raises:
            URINotSupported: If class not found
        """
        # Try local, stubs modules
        for module in [local, stubs]:
            if hasattr(module, class_name):
                return getattr(module, class_name)

        raise URINotSupported(f"Writer class not found: {class_name}")
    ```

    ## Overview

    The storage layer provides URI-driven access to artifact storage backends. Master can persist and retrieve checkpoints, metrics, and lineage data via URIs like:

    - `file:///path/to/file` — Local filesystem (implemented)
    - `s3://bucket/key` — AWS S3 (stub, raises NotImplementedError)

    - `worker://host/path` — Worker node (future)

    ## Architecture

    ### Components

    1. **BaseStorageWriter** — Abstract base class defining the storage interface
    2. **LocalStorageWriter** — File system implementation
    3. **S3StorageWriter, NFSStorageWriter** — Stub implementations
    4. **URIResolver** — Dispatcher that routes operations to correct backend
    5. **Error Hierarchy** — StorageError, FileNotFound, URINotSupported

    ### Data Flow

    ```
    Master API
         |
         v
    URIResolver.get_writer(uri)
         |
         +-- Extracts prefix (file://, s3://, etc.)
         +-- Looks up in STORAGE_BACKENDS config
         +-- Returns cached writer instance
         |
         v
    BaseStorageWriter (LocalStorageWriter, S3StorageWriter, ...)
         |
         v
    Backend (filesystem, S3, NFS, ...)
    ```

    from master.storage import URIResolver, LocalStorageWriter

    resolver = URIResolver()

    # Get writer for a URI
    writer = resolver.get_writer("file:///tmp/checkpoint.pt")

    # Write file
    await writer.write_file("file:///tmp/checkpoint.pt", data)

    # Read file
    data = await writer.read_file("file:///tmp/checkpoint.pt")

    # Check existence

    # Delete file
    await writer.delete_file("file:///tmp/checkpoint.pt")
