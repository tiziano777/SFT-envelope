```python
class SimulatedWorker:
    def __init__(self, master_uri: str, api_key: str, temp_dir: Path):
        self.master_uri = master_uri
        self.api_key = api_key
        self.setup_dir = temp_dir / "setup_simulated"
        self.state_file = self.setup_dir / ".worker_state.json"
        self.exp_id_file = self.setup_dir / ".exp_id"
        self.handshake_done_file = self.setup_dir / ".handshake_done"
        self.training_done_file = self.setup_dir / ".training_done"
        self.connection = None

    async def run_lifecycle(self, handshake_req: HandshakeRequest, strategy: Strategy) -> dict:
        """
        Run:
        1. Bootstrap (create setup_dir/.worker_state.json)
        2. Handshake (POST /handshake)
        3. Assert strategy matches expected
        4. Checkpoint push (POST /checkpoint_push)
        5. Status update: training_done (POST /status_update)
        6. Daemon flush (5s pause to allow async updates)
        7. Return experiment_id for verification

        Returns dict with:
        - exp_id: experiment ID from handshake response
        - strategy: Strategy returned from handshake
        - ckp_id: checkpoint ID from push response
        - status: final status from status_update
        """

    async def push_checkpoint(self, ckp_id: str, uri: str | None = None) -> dict:
        """Push checkpoint with optional uri=NULL for testing."""

    async def cleanup(self) -> None:
        """Remove setup_dir and state files."""
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Worker → Master | HandshakeRequest/CheckpointPush must be authenticated with valid X-API-Key |
| Master → Neo4j | Test fixture uses isolated Neo4j container (testcontainers), no production DB access |
| Test → Subprocess | SimulatedMaster subprocess runs with isolated env vars, no root access |
| Test nodes → Cleanup | _TEST label ensures test data is ephemeral; DETACH DELETE prevents data pollution |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-01 | Spoofing | Worker auth headers | mitigate | All test requests must include valid X-API-Key header (tests/integration/conftest.py fixture provides auth_headers) |
| T-09-02 | Tampering | Neo4j test data | mitigate | Test nodes use _TEST label, cleanup runs via DETACH DELETE in fixture teardown (auto) |
| T-09-03 | Repudiation | Event audit trail | accept | Event logging in transfer_log.jsonl for debugging; not required to be tamper-proof in test env |
| T-09-04 | Information Disclosure | Test env vars | mitigate | Master subprocess env vars include fake API keys ("test-api-key-12345"), no production secrets exposed |
| T-09-05 | Denial of Service | Daemon crash | mitigate | Test covers crash recovery; normal operations resume from last checkpoint (test_daemon_crash_recovery validates) |
| T-09-06 | Elevation of Privilege | Subprocess privileges | accept | SimulatedMaster subprocess runs as test user (no privilege escalation); acceptable for test environment |

</threat_model>
