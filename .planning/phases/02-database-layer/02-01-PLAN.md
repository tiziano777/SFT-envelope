```python
async def test_created_at_timestamp_on_node_create():

    await session.run("""
        CREATE (e:Experiment {exp_id: 'e-001', config_hash: 'h1', code_hash: 'h2', req_hash: 'h3'})

        RETURN e.created_at IS NOT NULL AS has_timestamp
    """)

    assert record["has_timestamp"] is True

async def test_orphan_checkpoint_validation_fails():
    """Creating Checkpoint without parent Experiment violates validation trigger."""
    with pytest.raises(Neo4jIntegrityError):
        await session.run("""
            CREATE (:Checkpoint {ckp_id: 'c-001', epoch: 1, run: 0})
        """)
```

**Implementation:**
- File: `master/neo4j/triggers.cypher`
- Trigger 1 (timestamps on create):
  ```cypher
  CALL apoc.trigger.install('neo4j', 'created_at_trigger',
    'UNWIND apoc.trigger.nodesByLabel($createdNodes, null) AS node
     SET node.created_at = datetime()',
    {phase: 'before'})
  ```
- Trigger 2 (timestamps on update):
  ```cypher
  CALL apoc.trigger.install('neo4j', 'updated_at_trigger',
    'UNWIND apoc.trigger.nodesByLabel($changedNodes, null) AS node
     SET node.updated_at = datetime()',
    {phase: 'before'})
  ```
- Trigger 3 (orphan validation):
  ```cypher
  CALL apoc.trigger.install('neo4j', 'orphan_checkpoint_validation',
    'UNWIND apoc.trigger.nodesByLabel($createdNodes, "Checkpoint") AS ckp
     WHERE NOT (ckp)-[:PRODUCED_BY|DERIVED_FROM|RETRY_FROM|MERGED_FROM]->()
       AND NOT ckp.is_merging = true
     CALL apoc.util.validate(false, "Checkpoint %s is orphan", [ckp.ckp_id])',
    {phase: 'after'})
  ```

**Implementation:**
- File: `master/neo4j/client.py`
- Module-level code:
  ```python
  from neo4j.asyncio import GraphDatabase, AsyncDriver
  import os

  _driver: AsyncDriver | None = None

  async def get_driver() -> AsyncDriver:
      global _driver
      if _driver is None:
          uri = os.getenv("NEO4J_URI", "bolt://localhost:7687")
          user = os.getenv("NEO4J_USER", "neo4j")
          password = os.getenv("NEO4J_PASSWORD", "password")
          pool_size = int(os.getenv("NEO4J_POOL_SIZE", "50"))

          _driver = GraphDatabase.driver(
              uri,
              auth=(user, password),
              max_pool_size=pool_size,
          )
      return _driver

  async def close_driver():
      global _driver
      if _driver:
          await _driver.close()
          _driver = None
  ```

- ABC class:
  ```python
  from abc import ABC, abstractmethod
  from envelope.middleware.shared import ExperimentNode, CheckpointNode, ConfigSnapshot

  class BaseExperimentRepository(ABC):
      """Abstract repository for experiment lineage operations."""

      @abstractmethod
      async def create_experiment(
          self,
          recipe_id: str,
          exp_id: str,
          model_id: str,
          config_hash: str,
          code_hash: str,
          req_hash: str,
          config_snapshot: ConfigSnapshot,
      ) -> ExperimentNode:
          """Create new experiment node (idempotent via MERGE)."""
          pass

      @abstractmethod
      async def upsert_checkpoint(
          self,
          exp_id: str,
          ckp_id: str,
          epoch: int,
          run: int,
          metrics_snapshot: dict,
          uri: str | None = None,
      ) -> CheckpointNode:
          """Create or update checkpoint (atomic transaction)."""
          pass

      @abstractmethod
      async def find_experiment_by_hashes(
          self,
          config_hash: str,
          code_hash: str,
          req_hash: str,
      ) -> ExperimentNode | None:
          """Find experiment matching all three hashes (handshake)."""
          pass

      @abstractmethod
      async def get_latest_checkpoint(self, exp_id: str) -> CheckpointNode | None:
          """Get most recent checkpoint for experiment (RESUME logic)."""
          pass

      @abstractmethod
      async def create_merged_checkpoint(
          self,
          exp_id: str,
          merged_ckp_id: str,
          source_ckp_ids: list[str],
          epoch: int,
      ) -> CheckpointNode:
          """Create checkpoint with N MERGED_FROM relations."""
          pass

      # ... additional methods (create_derived_from_relation, create_retry_from_relation, get_experiment)
  ```
