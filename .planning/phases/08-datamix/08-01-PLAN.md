```python
from pydantic import BaseModel, Field
from typing import Optional, List, Union

class DatamixSource(BaseModel):
    """Single data source in a datamix."""
    uri: str  # e.g., "s3://mybucket/dataset.parquet"
    replica: int = 1  # Number of times to repeat this source
    samples: int = 0  # 0 = all, else limit to N samples
    dist_name: Optional[str] = None  # Distribution name for sampling
    chat_type: str = "instruct"  # "instruct" or "multi_turn"

    class Config:
        json_schema_extra = {
            "example": {
                "uri": "s3://data/dataset-v1.parquet",
                "replica": 2,
                "samples": 10000,
                "chat_type": "instruct"
            }
        }

class DatamixConfig(BaseModel):
    """Multi-source dataset configuration."""
    sources: List[DatamixSource]

    class Config:
        json_schema_extra = {
            "example": {
                "sources": [
                    {"uri": "s3://data/dataset-v1.parquet", "replica": 1},
                    {"uri": "s3://data/dataset-v2.parquet", "replica": 2},
                ]
            }
        }

# Extend existing EnvelopeConfig
class EnvelopeConfig(BaseModel):
    """...(existing fields)"""

    # Existing single-dataset support (backward compat)
    dataset: Optional[str] = None

    # New: multi-source datamix support
    datamix: Optional[DatamixConfig] = None

    @property
    def has_datamix(self) -> bool:
        """Check if config uses datamix."""
        return self.datamix is not None and len(self.datamix.sources) > 0

    @property
    def has_single_dataset(self) -> bool:
        """Check if config uses single dataset (backward compat)."""
        return self.dataset is not None and not self.has_datamix

    class Config:
        validate_assignment = True
```

### Task 2: Create `envelope/prepare/datamix_loader.py` — Loader Logic

**Goal:** DatamixLoader detects and loads datamix vs single dataset

```python
# New file: envelope/prepare/datamix_loader.py
from pathlib import Path
from typing import Optional, List, Dict, Any
import pandas as pd
from envelope.config.models import EnvelopeConfig, DatamixConfig

class DatamixLoader:
    """Load and prepare datasets from config."""

    def __init__(self, setup_dir: Path):
        self.setup_dir = Path(setup_dir)
        self.config_file = setup_dir / "envelope" / "config.yaml"

    def load(self, config: EnvelopeConfig) -> pd.DataFrame:
        """
        Load dataset:
        - If datamix: load multiple sources with replica oversampling
        - Else: load single dataset (backward compat)
        """
        if config.has_datamix:
            return self._load_datamix(config.datamix)
        elif config.has_single_dataset:
            return self._load_single_dataset(config.dataset)
        else:
            raise ValueError("Config must have either 'dataset' or 'datamix' section")

    def _load_single_dataset(self, dataset_uri: str) -> pd.DataFrame:
        """Load single dataset (backward compat)."""
        print(f"Loading single dataset: {dataset_uri}")
        return self._load_from_uri(dataset_uri)

    def _load_datamix(self, datamix: DatamixConfig) -> pd.DataFrame:
        """Load and combine multiple sources with oversampling."""
        dfs = []

        for source in datamix.sources:
            print(f"Loading source: {source.uri} (replica={source.replica})")

            # Load from URI
            df = self._load_from_uri(source.uri)

            # Limit samples if specified
            if source.samples > 0:
                df = df.head(source.samples)

            # Apply replica oversampling
            if source.replica > 1:
                print(f"  Oversampling: {len(df)} rows × {source.replica} = {len(df) * source.replica}")
                df = pd.concat([df] * source.replica, ignore_index=True)

            dfs.append(df)

        # Combine all sources
        result = pd.concat(dfs, ignore_index=True)
        print(f"Combined dataset: {len(result)} rows from {len(dfs)} sources")

        return result

    def _load_from_uri(self, uri: str) -> pd.DataFrame:
        """Load parquet/json from URI."""
        if uri.startswith("s3://"):
            # S3 loading
            import s3fs
            with s3fs.S3File(uri, 'rb') as f:
                return pd.read_parquet(f)
        elif uri.startswith("file://"):
            # Local file
            path = Path(uri.replace("file://", ""))
            if uri.endswith(".parquet"):
                return pd.read_parquet(path)
            elif uri.endswith(".json"):
                return pd.read_json(path)
            else:
                raise ValueError(f"Unsupported format: {uri}")
        else:
            raise ValueError(f"Unsupported URI scheme: {uri}")
```

```python
# New file: tests/config/test_datamix.py
import pytest
from pathlib import Path
from envelope.config.models import EnvelopeConfig, DatamixConfig, DatamixSource
from envelope.prepare.datamix_loader import DatamixLoader
import pandas as pd

@pytest.fixture
def single_dataset_config(tmp_path):
    """Config with single dataset (backward compat)."""
    return EnvelopeConfig(
        dataset="s3://mybucket/dataset-v1.parquet",
        datamix=None
    )

@pytest.fixture
def datamix_config():
    """Config with datamix."""
    return EnvelopeConfig(
        datamix=DatamixConfig(sources=[
            DatamixSource(uri="file://data.parquet", replica=1, samples=100),
            DatamixSource(uri="file://data2.parquet", replica=2, samples=50),
        ])
    )

def test_single_dataset_backward_compat(single_dataset_config):
    """Single dataset config still works."""
    assert single_dataset_config.has_single_dataset
    assert not single_dataset_config.has_datamix
    assert single_dataset_config.dataset == "s3://mybucket/dataset-v1.parquet"

def test_datamix_config(datamix_config):
    """Datamix config is recognized."""
    assert datamix_config.has_datamix
    assert not datamix_config.has_single_dataset
    assert len(datamix_config.datamix.sources) == 2

def test_datamix_source_validation():
    """DatamixSource validates fields."""
    # Valid
    source = DatamixSource(uri="s3://bucket/data.parquet", replica=2)
    assert source.replica == 2

    # Invalid replica
    with pytest.raises(ValueError):
        DatamixSource(uri="s3://bucket/data.parquet", replica=0)

def test_datamix_replica_oversampling():
    """Replica field causes oversampling."""
    df_orig = pd.DataFrame({"a": [1, 2, 3]})

    # With replica=2, expect 6 rows
    df_replicated = pd.concat([df_orig] * 2, ignore_index=True)
    assert len(df_replicated) == 6

    # With replica=3, expect 9 rows
    df_replicated = pd.concat([df_orig] * 3, ignore_index=True)
    assert len(df_replicated) == 9

def test_loader_detects_datamix(tmp_path, datamix_config):
    """DatamixLoader detects datamix mode."""
    loader = DatamixLoader(tmp_path)

    assert datamix_config.has_datamix
    assert not datamix_config.has_single_dataset

def test_loader_detects_single_dataset(tmp_path, single_dataset_config):
    """DatamixLoader detects single dataset mode."""
    loader = DatamixLoader(tmp_path)

    assert single_dataset_config.has_single_dataset
    assert not single_dataset_config.has_datamix

def test_error_on_missing_dataset_section():
    """Error when neither dataset nor datamix is set."""
    config = EnvelopeConfig(dataset=None, datamix=None)

    with pytest.raises(ValueError, match="must have either"):
        loader = DatamixLoader(Path("/tmp"))
        loader.load(config)

def test_datamix_yaml_example():
    """Example datamix YAML config."""
    datamix = DatamixConfig(sources=[
        DatamixSource(
            uri="s3://company/llama-sft-v1.parquet",
            replica=1,
            samples=50000,
            chat_type="instruct"
        ),
        DatamixSource(
            uri="s3://company/gpt-conv-v2.parquet",
            replica=2,
            samples=25000,
            chat_type="multi_turn"
        ),
    ])

    # This represents:
    # - 50,000 rows from source 1 (1x)
    # - 50,000 rows from source 2 (2x25,000)
    # = 100,000 total rows
    total_samples = (50000 * 1) + (25000 * 2)
    assert total_samples == 100000
```
