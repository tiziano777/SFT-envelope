# New file: envelope/middleware/worker/watcher.py
import asyncio
from pathlib import Path
from typing import Callable, Set
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler, FileModifiedEvent

class WorkerFileHandler(FileSystemEventHandler):
    """Watch lineage/, training_metrics/, config/rewards/ directories."""

    def __init__(self, queue: asyncio.Queue, watched_dirs: Set[Path]):
        self.queue = queue
        self.watched_dirs = watched_dirs

    def on_modified(self, event: FileModifiedEvent):
        """Queue file change events."""
        if event.is_directory:
            return
        path = Path(event.src_path)
        if any(path.is_relative_to(d) for d in self.watched_dirs if d.exists()):
            asyncio.run_coroutine_threadsafe(
                self.queue.put({"type": "file_modified", "path": str(path)}),
                asyncio.get_event_loop()
            )

class WorkerWatcher:
    """Filesystem watcher for Worker setup."""

    def __init__(self, setup_dir: Path):
        self.setup_dir = Path(setup_dir)
        self.watched_dirs = {
            self.setup_dir / "lineage" / "to_transfer",
            self.setup_dir / "training_metrics",
            self.setup_dir / "config" / "rewards"
        }
        self.observer = None
        self.queue = asyncio.Queue()

    async def start(self):
        """Start watching directories."""
        handler = WorkerFileHandler(self.queue, self.watched_dirs)
        self.observer = Observer()
        for d in self.watched_dirs:
            d.mkdir(parents=True, exist_ok=True)
            self.observer.schedule(handler, str(d), recursive=True)
        self.observer.start()

    def stop(self):
        """Stop watching."""
        if self.observer:
            self.observer.stop()
            self.observer.join()

    async def get_event(self):
        """Yield filesystem events from queue."""
        return await self.queue.get()

### Task 2: Create `middleware/worker/pusher.py` — AsyncPusher with Retry

```python
# New file: envelope/middleware/worker/pusher.py
import asyncio
import json
from pathlib import Path
from typing import Set, Optional
from datetime import datetime, timedelta
import logging

from ..shared.connection import BaseConnection
from ..shared.models import TransferLogEntry

logger = logging.getLogger(__name__)

class AsyncPusher:
    """Queue-based async event pusher with exponential backoff retry."""

    def __init__(self,
                 connection: BaseConnection,
                 transfer_log: Path,
                 max_retries: int = 5,
                 initial_backoff: float = 1.0):
        self.connection = connection
        self.transfer_log = transfer_log
        self.max_retries = max_retries
        self.initial_backoff = initial_backoff
        self.queue = asyncio.Queue()
        self.seen_ids: Set[str] = set()  # Deduplication
        self._running = False

    async def push_event(self, event_id: str, event_type: str, data: dict) -> None:
        """Deduplicate and queue event for pushing."""
        if event_id in self.seen_ids:
            logger.debug(f"Skipping duplicate event {event_id}")
            return

        self.seen_ids.add(event_id)
        await self.queue.put({
            "event_id": event_id,
            "event_type": event_type,
            "data": data,
            "retry_count": 0
        })

    async def start(self) -> None:
        """Start pusher worker loop."""
        self._running = True
        while self._running:
            try:
                item = await asyncio.wait_for(self.queue.get(), timeout=1.0)
            except asyncio.TimeoutError:
                continue

            await self._push_with_retry(item)

    async def _push_with_retry(self, item: dict) -> None:
        """Push event with exponential backoff retry."""
        event_id = item["event_id"]
        retry_count = item["retry_count"]

        for attempt in range(self.max_retries):
            if attempt > 0:
                backoff = self.initial_backoff * (2 ** (attempt - 1))
                logger.info(f"Retry {attempt}/{self.max_retries} for {event_id} (backoff {backoff}s)")
                await asyncio.sleep(backoff)

            try:
                success = await self.connection.send_metadata(item["data"])
                if success:
                    self._log_transfer(event_id, item["event_type"], "sent")
                    return
            except Exception as e:
                logger.warning(f"Push attempt {attempt} failed: {e}")

        # Exhausted retries
        self._log_transfer(event_id, item["event_type"], "failed", error=str(Exception))

    def _log_transfer(self,
                      event_id: str,
                      event_type: str,
                      status: str,
                      error: Optional[str] = None) -> None:
        """Append to transfer log."""
        entry = TransferLogEntry(
            event_id=event_id,
            event_type=event_type,
            status=status,
            error=error
        )
        with open(self.transfer_log, 'a') as f:
            f.write(entry.model_dump_json() + '\n')

    async def stop(self) -> None:
        """Flush queue and stop."""
        self._running = False

        # Flush remaining items
        while not self.queue.empty():
            try:
                item = self.queue.get_nowait()
                await self._push_with_retry(item)
            except asyncio.QueueEmpty:
                break
```

### Task 3: Create `middleware/worker/worker.py` — Orchestration

```python
# New file: envelope/middleware/worker/worker.py
import asyncio
from pathlib import Path
from typing import Optional

from .daemon import WorkerDaemon
from .watcher import WorkerWatcher
from .pusher import AsyncPusher
from .http_connection import HTTPConnection

class Worker:
    """Orchestrate daemon, watcher, and pusher."""

    def __init__(self, setup_dir: Path, master_uri: str = "http://localhost:8000", api_key: str = ""):
        self.setup_dir = Path(setup_dir)
        self.daemon = WorkerDaemon(setup_dir, master_uri, api_key)
        self.watcher = WorkerWatcher(setup_dir)
        self.connection = HTTPConnection(master_uri, api_key)
        self.pusher = AsyncPusher(
            self.connection,
            self.setup_dir / "worker" / "transfer_log.jsonl"
        )
        self._tasks = []

    async def run(self) -> None:
        """Handshake, watch, and push events."""
        # Blocking handshake
        state = await self.daemon.bootstrap()
        if not state:
            print("Handshake failed — exiting")
            return

        print(f"Handshake succeeded: exp_id={state.exp_id}")

        # Start watcher and pusher
        await self.watcher.start()

        # Create background tasks
        pusher_task = asyncio.create_task(self.pusher.start())
        watcher_task = asyncio.create_task(self._watch_loop())

        self._tasks = [pusher_task, watcher_task]

        try:
            await asyncio.gather(*self._tasks)
        except KeyboardInterrupt:
            await self.shutdown()

    async def _watch_loop(self) -> None:
        """Convert filesystem events to push events."""
        while True:
            try:
                event = await asyncio.wait_for(self.watcher.get_event(), timeout=1.0)
                # Parse event and push
                event_id = f"{event['path']}-{int(asyncio.get_running_loop().time())}"
                await self.pusher.push_event(
                    event_id=event_id,
                    event_type="filechange",
                    data=event
                )
            except asyncio.TimeoutError:
                continue

    async def shutdown(self) -> None:
        """Graceful shutdown."""
        print("Shutting down worker...")
        self.watcher.stop()
        await self.pusher.stop()
        for task in self._tasks:
            task.cancel()
```
