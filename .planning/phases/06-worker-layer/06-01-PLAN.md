### Task 2: Create `middleware/worker/daemon.py` — Bootstrap & Handshake

**Goal:** Daemon lifecycle management and blocking handshake

**File:** `envelope/middleware/worker/daemon.py`

```python
# New file
import asyncio
import os
from pathlib import Path
from typing import Optional
from datetime import datetime
import httpx

from envelope.shared.models import (
    HandshakeRequest, HandshakeResponse, Strategy,
    ConfigSnapshot
)
from ..shared.models import WorkerState, TransferLogEntry

class WorkerDaemon:
    """Worker daemon: handshake, persist state, prepare for event pushing."""

    def __init__(self,
                 setup_dir: Path,
                 master_uri: str = "http://localhost:8000",
                 api_key: str = "",
                 handshake_timeout: int = 30):
        self.setup_dir = Path(setup_dir)
        self.master_uri = master_uri
        self.api_key = api_key
        self.handshake_timeout = handshake_timeout
        self.state_file = self.setup_dir / "worker" / ".worker_state.json"
        self.exp_id_file = self.setup_dir / "worker" / ".exp_id"
        self.handshake_done_file = self.setup_dir / "worker" / ".handshake_done"
        self.transfer_log = self.setup_dir / "worker" / "transfer_log.jsonl"

    async def bootstrap(self) -> Optional[WorkerState]:
        """Perform blocking handshake and persist state."""
        try:
            state = await self._handshake()
            if state:
                self._persist_state(state)
                return state
            else:
                print("Handshake failed — entering degraded mode")
                return None
        except asyncio.TimeoutError:
            print(f"Handshake timeout ({self.handshake_timeout}s) — degraded mode")
            return None

    async def _handshake(self) -> Optional[WorkerState]:
        """Blocking handshake with Master."""
        config_snapshot = ConfigSnapshot.from_setup_dir(self.setup_dir)
        request = HandshakeRequest(
            recipe_id="...",
            config=config_snapshot,
            ...
        )

        try:
            async with httpx.AsyncClient() as client:
                resp = await asyncio.wait_for(
                    client.post(
                        f"{self.master_uri}/handshake",
                        json=request.model_dump(),
                        headers={"X-API-Key": self.api_key},
                        timeout=self.handshake_timeout
                    ),
                    timeout=self.handshake_timeout
                )
                response = HandshakeResponse(**resp.json())

                return WorkerState(
                    exp_id=response.exp_id,
                    recipe_id=request.recipe_id,
                    master_uri=self.master_uri,
                    strategy=response.strategy.value,
                )
        except Exception as e:
            print(f"Handshake error: {e}")
            return None

    def _persist_state(self, state: WorkerState):
        """Atomically persist state (tmp + rename)."""
        tmp_file = self.state_file.with_suffix('.tmp')
        tmp_file.write_text(state.model_dump_json())
        tmp_file.rename(self.state_file)

        # Write .exp_id
        self.exp_id_file.write_text(state.exp_id)

        # Write .handshake_done marker
        self.handshake_done_file.touch()

    def log_transfer(self, entry: TransferLogEntry):
        """Append transfer log entry."""
        self.transfer_log.parent.mkdir(parents=True, exist_ok=True)
        with open(self.transfer_log, 'a') as f:
            f.write(entry.model_dump_json() + '\n')
```

**Acceptance Criteria:**
- ✓ Blocking handshake with timeout (default 30s)
- ✓ Write .handshake_done, .exp_id, .worker_state.json atomically
- ✓ Fallback to degraded mode on timeout/error
- ✓ transfer_log.jsonl grows append-only
- ✓ All async awaits use asyncio.wait_for with timeout

### Task 3: Create `middleware/shared/state.py` — Atomic Persistence Utils

**Goal:** Reusable atomic write/read patterns

**File:** `envelope/middleware/shared/state.py`

```python
# New file
from pathlib import Path
import json
import tempfile
from typing import TypeVar, Generic

T = TypeVar('T')

class AtomicStateManager(Generic[T]):
    """Atomic file write using tmp + rename pattern."""

    def save(self, path: Path, obj: dict) -> None:
        """Save JSON atomically."""
        with tempfile.NamedTemporaryFile(
            dir=path.parent,
            mode='w',
            delete=False,
            suffix='.tmp'
        ) as tmp:
            json.dump(obj, tmp)
            tmp_path = Path(tmp.name)
        tmp_path.rename(path)

    def load(self, path: Path) -> Optional[dict]:
        """Load JSON or return None."""
        if path.exists():
            return json.loads(path.read_text())
        return None
```

**Acceptance Criteria:**
- ✓ Never leaves partial files on disk
- ✓ Works with Path objects
- ✓ Reusable across middleware

import pytest
import asyncio
from pathlib import Path
from unittest.mock import AsyncMock, patch, MagicMock
from datetime import datetime

from envelope.middleware.worker.daemon import WorkerDaemon
from envelope.middleware.shared.models import WorkerState, TransferLogEntry

@pytest.fixture
def setup_dir(tmp_path):
    (tmp_path / "worker").mkdir()
    return tmp_path

@pytest.mark.asyncio
async def test_bootstrap_success(setup_dir):
    """Handshake succeeds, state persisted."""
    daemon = WorkerDaemon(setup_dir, handshake_timeout=5)

    with patch.object(daemon, '_handshake', new_callable=AsyncMock) as mock_hs:
        mock_hs.return_value = WorkerState(
            exp_id="exp-123",
            recipe_id="recipe-abc",
            master_uri="http://localhost:8000",
            strategy="NEW"
        )

        state = await daemon.bootstrap()

        assert state is not None
        assert state.exp_id == "exp-123"
        assert (setup_dir / "worker" / ".exp_id").read_text() == "exp-123"
        assert (setup_dir / "worker" / ".handshake_done").exists()
        mock_hs.assert_called_once()

@pytest.mark.asyncio
async def test_bootstrap_timeout(setup_dir):
    """Timeout → degraded mode."""
    daemon = WorkerDaemon(setup_dir, handshake_timeout=1)

    with patch.object(daemon, '_handshake', new_callable=AsyncMock) as mock_hs:
        async def timeout_coro():
            await asyncio.sleep(2)
        mock_hs.side_effect = asyncio.TimeoutError()

        state = await daemon.bootstrap()

        assert state is None

@pytest.mark.asyncio
async def test_transfer_log_append(setup_dir):
    """Transfer log appends without duplicates."""
    daemon = WorkerDaemon(setup_dir)

    entry1 = TransferLogEntry(
        event_id="ev-1",
        event_type="checkpoint",
        status="sent"
    )
    entry2 = TransferLogEntry(
        event_id="ev-2",
        event_type="metric",
        status="pending"
    )

    daemon.log_transfer(entry1)
    daemon.log_transfer(entry2)

    lines = daemon.transfer_log.read_text().strip().split('\n')
    assert len(lines) == 2
    assert lines[0].startswith('{"event_id":"ev-1"')
    assert lines[1].startswith('{"event_id":"ev-2"')
